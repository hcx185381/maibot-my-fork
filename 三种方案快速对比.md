```bash

copy docker-config\mmc\model_config.toml docker-config\mmc\model_config.toml.backup

# 2. 使用全免费配置
copy docker-config\mmc\model_config_siliconflow_free_only.toml docker-config\mmc\model_config.toml

# 3. 替换API Key（可选，如果需要）
# 编辑 model_config.toml，替换第18行的API Key

# 4. 重启容器

# 5. 测试对话
```
